import java.util.*;
import java.io.*;

public class Main{
    public static void main(String[] args) throws Exception{
        Scanner sc=new Scanner(System.in);
        while(sc.hasNext()){
            int n=sc.nextInt();
            int x=sc.nextInt();
            System.out.println(calculate(n,x));
        }
        sc.close();
    }
    
    public static String calculate(int n,int x){
        if(6*n<x){
            return "0";
        }
        if(x<=n){
            return "1";
        }
        //dp[i][j]表示i个骰子和为j有多少种组合
        //dp[i][j]=dp[i-1][j-k]之和（k=1~6）
        long[][] dp=new long[30][200];
        //初始化数组
        for(int i=0;i<=29;i++){
            for(int j=0;j<=199;j++){
                dp[i][j]=0;
            }
        }
        for(int i=1;i<=n;i++){
            for(int j=i;j<=6*i;j++){
                //特殊情况直接赋1
                if(i==1||i==j||j==i*6){
                    dp[i][j]=1;
                }
                //从i-1的情况算出结果
                else{
                    for(int k=1;k<=6;k++){
                        //限定范围，防止数组越界
                        if(j-k>=i-1) {
                            dp[i][j]+=dp[i-1][j-k];
                        }
                    }
                }
            }
        }
        //统计组合数
        long result=0;
        for(int i=x;i<=6*n;i++){
            result+=dp[n][i];
        }
        long total=1;
        for(int i=1;i<=n;i++){
            total*=6;
        }
        //化简分数
        while(true){
            if(result%2==0&&total%2==0){
                result/=2;
                total/=2;
            }
            else if(result%3==0&&total%3==0){
                result/=3;
                total/=3;
            }
            else{
                break;
            }
        }
        return String.valueOf(result)+"/"+String.valueOf(total);
    }
}
