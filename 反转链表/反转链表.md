public class Solution {
    public ListNode ReverseList(ListNode head) {
        if(head==null){
            return null;
        }
        ListNode now=head;
        ListNode pre=null;
        ListNode next=now.next;
        for(;;){
            if(next==null){
                now.next=pre;
                break;
            }
            else{
                now.next=pre;
                pre=now;
                now=next;
                next=now.next;
            }
        }
        return now;
    }
}
