import java.util.*;
import java.io.*;
public class Main{
    public static void main(String[] args) throws Exception{
        Scanner sc = new Scanner(System.in);
        while(sc.hasNext()){
            String str1=sc.nextLine();
            String str2=sc.nextLine();
            System.out.println(encrypt(str1));
            System.out.println(unencrypt(str2));
        }
        sc.close();
    }
    
    //char可以用ASCII码的值进行比较和计算
    public static String encrypt(String s){
        char[] array=s.toCharArray();
        StringBuilder sb=new StringBuilder();
        for(char c:array){
            if(c>='0'&&c<='8'){
                c++;
            }
            else if(c=='9'){
                c='0';
            }
            else if(c>='A'&&c<='Y'){
                c=(char)(c+33);
            }
            else if(c=='Z'){
                c='a';
            }
            else if(c>='a'&&c<='y'){
                c=(char)(c-31);
            }
            else if(c=='z'){
                c='A';
            }
            sb.append(c);
        }
        return sb.toString();
    }
    
    public static String unencrypt(String s){
        char[] array=s.toCharArray();
        StringBuilder sb=new StringBuilder();
        for(char c:array){
            if(c>='1'&&c<='9'){
                c--;
            }
            else if(c=='0'){
                c='9';
            }
            else if(c>='B'&&c<='Z'){
                c=(char)(c+31);
            }
            else if(c=='A'){
                c='z';
            }
            else if(c>='b'&&c<='z'){
                c=(char)(c-33);
            }
            else if(c=='a'){
                c='Z';
            }
            sb.append(c);
        }
        return sb.toString();
    }
}
