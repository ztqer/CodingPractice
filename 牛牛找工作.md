import java.util.*;
public class Main{
    public static void main(String[] args){
        Scanner sc=new Scanner(System.in);
        int N=sc.nextInt();
        int M=sc.nextInt();
        int t1=0,t2=0;
        HashMap<Integer,Integer> hs = new HashMap<>();
        int[] a = new int[N+M];
        int[] b = new int[M];
        for(int i=0;i<N;i++){
            t1 = sc.nextInt();
            t2 = sc.nextInt();
            a[i] = t1;
            if(hs.containsKey(t1)) {
                if(hs.get(t1)>=t2) {
                    continue;
                }
            }
            hs.put(t1,t2);
        }
        for(int i=0;i<M;i++){
            t1 = sc.nextInt();
            a[N+i] = t1;
            b[i] = t1;
            if(!hs.containsKey(t1))
               hs.put(t1,0);
        }
        Arrays.sort(a);
        int max = 0;
        for(int i=0;i<N+M;i++){
            max = Math.max(max,hs.get(a[i]));
            hs.put(a[i],max);
        }
        for(int i=0;i<M;i++){
            System.out.println(hs.get(b[i]));
        }
    }
}
